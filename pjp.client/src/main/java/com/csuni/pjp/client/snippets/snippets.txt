class HandListCell extends ListCell<CardModel> {
    private ImageView imageView = new ImageView();
    private GameViewModel vm;


    public HandListCell(GameViewModel vm) {
        super();
        this.vm = vm;
        this.imageView.setScaleX(0.4);
        this.imageView.setScaleY(0.4);
        setOnMouseClicked(event -> vm.placeAction(getItem()));
    }
    
    @Override
    public void updateItem(CardModel name, boolean empty) {
        super.updateItem(name, empty);
        if(empty) {
            setText(null);
            setGraphic(null);
        }
        else {
            imageView.setImage(vm.cardImages.getCardImages().get(name.getSuit()).get(name.getRank()));
            setGraphic(imageView);
        }
    }
}
